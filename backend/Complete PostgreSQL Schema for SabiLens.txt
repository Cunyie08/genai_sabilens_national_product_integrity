1. Users & Authentication Schema


```sql
-- Users table (all roles: consumer, company_admin, nafdac_officer, nafdac_admin) CREATE TABLE users (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), email VARCHAR(255) UNIQUE,
phone VARCHAR(20) UNIQUE NOT NULL,
password_hash VARCHAR(255) NOT NULL, first_name VARCHAR(100),
last_name VARCHAR(100),
  role VARCHAR(50) NOT NULL CHECK (role IN ('consumer', 'company_admin', 'nafdac_officer', 'nafdac_admin')),
  status VARCHAR(50) DEFAULT 'active' CHECK (status IN ('active', 'suspended', 'pending', 'deleted')),
email_verified BOOLEAN DEFAULT FALSE, phone_verified BOOLEAN DEFAULT FALSE, profile_image_url TEXT,

preferred_language VARCHAR(10) DEFAULT 'en', last_login TIMESTAMP,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
deleted_at TIMESTAMP
);


-- Create index for faster lookups
CREATE INDEX idx_users_email ON users(email) WHERE deleted_at IS NULL; CREATE INDEX idx_users_phone ON users(phone) WHERE deleted_at IS NULL; CREATE INDEX idx_users_role ON users(role);

-- Sessions table
CREATE TABLE sessions (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
refresh_token TEXT NOT NULL, device_info JSONB,
ip_address INET,
expires_at TIMESTAMP NOT NULL,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE INDEX idx_sessions_user_id ON sessions(user_id);
CREATE INDEX idx_sessions_refresh_token ON sessions(refresh_token);


-- OTP verifications
CREATE TABLE otp_verifications (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), user_id UUID REFERENCES users(id) ON DELETE CASCADE, phone VARCHAR(20),
email VARCHAR(255),
otp_code VARCHAR(6) NOT NULL,

  purpose VARCHAR(50) CHECK (purpose IN ('registration', 'login', 'password_reset', 'phone_verification')),
expires_at TIMESTAMP NOT NULL, verified_at TIMESTAMP,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
CONSTRAINT at_least_one_contact CHECK (phone IS NOT NULL OR email IS NOT NULL)
);


CREATE INDEX idx_otp_phone ON otp_verifications(phone); CREATE INDEX idx_otp_email ON otp_verifications(email);
```



2. Companies Schema


```sql
-- Companies table
CREATE TABLE companies (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), name VARCHAR(255) NOT NULL,
registration_number VARCHAR(100) UNIQUE NOT NULL, email VARCHAR(255) UNIQUE NOT NULL,
phone VARCHAR(20) NOT NULL,
address TEXT,
city VARCHAR(100), state VARCHAR(100),
country VARCHAR(100) DEFAULT 'Nigeria', logo_url TEXT,
website VARCHAR(255),
  status VARCHAR(50) DEFAULT 'pending' CHECK (status IN ('pending', 'active', 'suspended', 'rejected')),
  subscription_tier VARCHAR(50) DEFAULT 'free' CHECK (subscription_tier IN ('free', 'basic', 'premium', 'enterprise')),
subscription_expires_at TIMESTAMP,

verified_at TIMESTAMP,
verified_by UUID REFERENCES users(id),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE INDEX idx_companies_status ON companies(status);
CREATE INDEX idx_companies_registration ON companies(registration_number);


-- Company team members (users with company_admin role)
ALTER TABLE users ADD COLUMN company_id UUID REFERENCES companies(id) ON DELETE SET NULL;


-- Company settings
CREATE TABLE company_settings (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE, notification_email BOOLEAN DEFAULT TRUE,
notification_alerts BOOLEAN DEFAULT TRUE, notification_reports BOOLEAN DEFAULT TRUE, auto_export_enabled BOOLEAN DEFAULT FALSE, api_access_enabled BOOLEAN DEFAULT FALSE, webhook_url TEXT,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
UNIQUE(company_id)
);


-- Company API keys
CREATE TABLE company_api_keys (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE, api_key VARCHAR(255) UNIQUE NOT NULL,

api_secret VARCHAR(255) NOT NULL, name VARCHAR(100) NOT NULL,
permissions JSONB DEFAULT '[]', last_used_at TIMESTAMP, expires_at TIMESTAMP,
is_active BOOLEAN DEFAULT TRUE,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE INDEX idx_api_keys_company ON company_api_keys(company_id);

-- Subscription plans
CREATE TABLE subscription_plans (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
name VARCHAR(100) NOT NULL,
tier VARCHAR(50) NOT NULL UNIQUE CHECK (tier IN ('free', 'basic', 'premium', 'enterprise')),
description TEXT,
price_monthly DECIMAL(10,2),
price_annual DECIMAL(10,2),
features JSONB DEFAULT '{}',
max_scans_monthly INTEGER,
max_products INTEGER,
max_team_members INTEGER,
api_access BOOLEAN DEFAULT FALSE,
webhook_support BOOLEAN DEFAULT FALSE,
priority_support BOOLEAN DEFAULT FALSE,
is_active BOOLEAN DEFAULT TRUE,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_subscription_plans_tier ON subscription_plans(tier);

-- Webhook logs (for tracking webhook deliveries)
CREATE TABLE webhook_logs (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
webhook_url TEXT NOT NULL,
event_type VARCHAR(100),
payload JSONB,
response_status INTEGER,
response_body TEXT,
retry_count INTEGER DEFAULT 0,
last_retry_at TIMESTAMP,
success BOOLEAN DEFAULT FALSE,
error_message TEXT,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_webhook_logs_company ON webhook_logs(company_id);
CREATE INDEX idx_webhook_logs_created ON webhook_logs(created_at);
```



3. Products Registry


```sql
-- Product categories
CREATE TABLE product_categories (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(), name VARCHAR(100) NOT NULL,
description TEXT,
parent_id UUID REFERENCES product_categories(id), is_active BOOLEAN DEFAULT TRUE,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


-- Products
CREATE TABLE products (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
company_id UUID NOT NULL REFERENCES companies(id) ON DELETE CASCADE, name VARCHAR(255) NOT NULL,
description TEXT,
category_id UUID REFERENCES product_categories(id),

nafdac_number VARCHAR(100) UNIQUE NOT NULL, batch_number VARCHAR(100),
manufacturer_name VARCHAR(255), manufacturing_date DATE, expiry_date DATE,
ingredients TEXT, warnings TEXT, image_urls TEXT[], master_artwork_url TEXT,
  status VARCHAR(50) DEFAULT 'active' CHECK (status IN ('active', 'pending', 'expired', 'discontinued')),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
created_by UUID REFERENCES users(id)
);


CREATE INDEX idx_products_company ON products(company_id); CREATE INDEX idx_products_nafdac ON products(nafdac_number); CREATE INDEX idx_products_status ON products(status);

-- Product barcodes
CREATE TABLE product_barcodes (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
product_id UUID NOT NULL REFERENCES products(id) ON DELETE CASCADE,
barcode_type VARCHAR(50) CHECK (barcode_type IN ('UPC', 'EAN', 'QR', 'CODE128', 'OTHER')),
barcode_value VARCHAR(255) NOT NULL, is_primary BOOLEAN DEFAULT FALSE,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
UNIQUE(barcode_type, barcode_value)
);

-- Product embeddings (AI model fingerprints for similarity matching)
CREATE TABLE product_embeddings (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
product_id UUID NOT NULL REFERENCES products(id) ON DELETE CASCADE,
embedding_type VARCHAR(50) NOT NULL CHECK (embedding_type IN ('visual', 'ocr', 'regulatory', 'fusion', 'multi_modal')),
embedding_vector BYTEA NOT NULL,
model_version VARCHAR(50),
confidence_score DECIMAL(5,2),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_product_embeddings_product ON product_embeddings(product_id);
CREATE INDEX idx_product_embeddings_type ON product_embeddings(embedding_type);

4. Scan History & Reports


```sql
-- Scans table
CREATE TABLE scans (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
product_id UUID REFERENCES products(id), product_name VARCHAR(255),
category VARCHAR(100), manufacturer VARCHAR(255), nafdac_number VARCHAR(100), batch_number VARCHAR(100), expiry_date DATE,
status VARCHAR(50) CHECK (status IN ('authentic', 'caution', 'fake', 'pending')), similarity_score DECIMAL(5,2),
confidence_score DECIMAL(5,2), scan_image_urls TEXT[], scan_location GEOGRAPHY(POINT), location_name VARCHAR(255), scan_metadata JSONB,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


CREATE INDEX idx_scans_user ON scans(user_id); CREATE INDEX idx_scans_product ON scans(product_id); CREATE INDEX idx_scans_status ON scans(status); CREATE INDEX idx_scans_created ON scans(created_at);
CREATE INDEX idx_scans_location ON scans USING GIST(scan_location);


-- Scan analysis results (detailed AI output) CREATE TABLE scan_analyses (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),

scan_id UUID NOT NULL REFERENCES scans(id) ON DELETE CASCADE,
ai_model_version VARCHAR(50), features_detected JSONB, anomalies JSONB,
ocr_text TEXT, visual_fingerprint JSONB, analysis_time_ms INTEGER,
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


-- Counterfeit reports CREATE TABLE reports (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
scan_id UUID NOT NULL REFERENCES scans(id) ON DELETE CASCADE,
user_id UUID NOT NULL REFERENCES users(id), product_id UUID REFERENCES products(id), company_id UUID REFERENCES companies(id),
  report_status VARCHAR(50) DEFAULT 'pending' CHECK (report_status IN ('pending', 'verified', 'investigating', 'resolved', 'rejected')),
severity VARCHAR(50) CHECK (severity IN ('critical', 'high', 'medium', 'low')),
  purchase_channel VARCHAR(50) CHECK (purchase_channel IN ('market_stall', 'roadside_vendor', 'supermarket', 'supplier', 'online', 'other')),
purchase_channel_details TEXT, price_paid DECIMAL(10,2), receipt_image_urls TEXT[], additional_comments TEXT, gps_location GEOGRAPHY(POINT), location_name VARCHAR(255),
reported_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
resolved_at TIMESTAMP,
resolved_by UUID REFERENCES users(id), resolution_notes TEXT
);

CREATE INDEX idx_reports_scan ON reports(scan_id);
CREATE INDEX idx_reports_company ON reports(company_id); CREATE INDEX idx_reports_status ON reports(report_status); CREATE INDEX idx_reports_severity ON reports(severity);
CREATE INDEX idx_reports_location ON reports USING GIST(gps_location);


-- Report evidence
CREATE TABLE report_evidence (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
report_id UUID NOT NULL REFERENCES reports(id) ON DELETE CASCADE,
evidence_type VARCHAR(50) CHECK (evidence_type IN ('image', 'video', 'receipt', 'document')), file_url TEXT NOT NULL,
file_size INTEGER, mime_type VARCHAR(100),
uploaded_by UUID REFERENCES users(id),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
```

5.1 NAFDAC Cases & Enforcement


```sql
-- NAFDAC Cases (investigation records)
CREATE TABLE cases (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
case_number VARCHAR(100) UNIQUE NOT NULL,
report_id UUID REFERENCES reports(id) ON DELETE SET NULL,
assigned_to UUID REFERENCES users(id),
case_status VARCHAR(50) NOT NULL DEFAULT 'pending' CHECK (case_status IN ('pending', 'investigating', 'under_review', 'suspended', 'closed', 'escalated')),
priority VARCHAR(50) DEFAULT 'medium' CHECK (priority IN ('critical', 'high', 'medium', 'low')),
description TEXT,
officer_notes TEXT,
location GEOGRAPHY(POINT),
location_name VARCHAR(255),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
closed_at TIMESTAMP
);

CREATE INDEX idx_cases_status ON cases(case_status);
CREATE INDEX idx_cases_assigned_to ON cases(assigned_to);
CREATE INDEX idx_cases_report ON cases(report_id);
CREATE INDEX idx_cases_created ON cases(created_at);

-- Enforcement actions (raids, seizures, notices)
CREATE TABLE enforcement_actions (
id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
case_id UUID NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
action_type VARCHAR(50) NOT NULL CHECK (action_type IN ('raid', 'seizure', 'notice', 'suspension', 'prosecution', 'other')),
description TEXT,
location GEOGRAPHY(POINT),
location_name VARCHAR(255),
executed_by UUID REFERENCES users(id),
execution_date TIMESTAMP,
quantity_seized INTEGER,
seized_items_description TEXT,
evidence_files TEXT[],
action_status VARCHAR(50) DEFAULT 'pending' CHECK (action_status IN ('pending', 'in_progress', 'completed', 'failed', 'cancelled')),
created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE INDEX idx_enforcement_case ON enforcement_actions(case_id);
CREATE INDEX idx_enforcement_type ON enforcement_actions(action_type);
CREATE INDEX idx_enforcement_status ON enforcement_actions(action_status);
CREATE INDEX idx_enforcement_location ON enforcement_actions USING GIST(location);
```

10. Relationships Summary


```sql
-- Users belong to Companies (if company_admin)
-- Companies have many Products
-- Users have many Scans
-- Scans generate Reports
-- Reports belong to Companies and Products
-- Reports generate Alerts for Companies and NAFDAC
-- Reports create Vendor Locations
-- Locations create Hotspots
-- All actions are Audited
